function updateTable(data) { const tbody = document.getElementById('dataTableBody'); tbody.innerHTML = ''; // FILTER KHUSUS TABEL: Hanya ambil DSR yang Active == 'YES' const activeTableData = data.filter(row => { const active = (row['Active'] || '').toString().trim().toUpperCase(); return active === 'YES'; }); if (activeTableData.length === 0) { tbody.innerHTML = `Tidak ada DSR active ditemukan.`; return; } activeTableData.forEach(row => { const name = row['Sales Name'] || '-'; const salesId = row['Sales ID'] || ''; const workplace = row['Workplace Name'] || '-'; const targetJul = cleanNum(row['Target Jul 26']); const actualJul = cleanNum(row['Actual Jul 26']); const mAchieve = targetJul > 0 ? ((actualJul / targetJul) * 100).toFixed(0) : cleanNum(row['% Monthly Achievement']); const prevMonth = cleanNum(row['Prev Month']); const momGrowth = prevMonth > 0 ? (((actualJul - prevMonth) / prevMonth) * 100).toFixed(0) : cleanNum(row['MoM Growth']); const targetYtd = cleanNum(row['Target YTD']); const actualYtd = cleanNum(row['Actual YTD']); const ytdAchieve = targetYtd > 0 ? ((actualYtd / targetYtd) * 100).toFixed(0) : cleanNum(row['% YTD Achievement']); const prevYtd = cleanNum(row['Prev YTD (L)']); const ytdGrowth = prevYtd > 0 ? (((actualYtd - prevYtd) / prevYtd) * 100).toFixed(0) : cleanNum(row['YTD Growth']); const tr = document.createElement('tr'); tr.className = 'hover:bg-slate-700/30 transition'; tr.innerHTML = `
${name}
${salesId} | ${workplace}
YES ${formatNum(targetJul)} ${formatNum(actualJul)} ${mAchieve}% ${formatNum(prevMonth)} ${momGrowth >= 0 ? '+' : ''}${momGrowth}% ${formatNum(targetYtd)} ${formatNum(actualYtd)} ${ytdAchieve}% ${formatNum(prevYtd)} ${ytdGrowth >= 0 ? '+' : ''}${ytdGrowth}% `; tbody.appendChild(tr); }); }
